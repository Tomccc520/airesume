```typescript
'use client'

import { useState, useEffect, useCallback, useMemo } from 'react'
import { useUndoRedo, useUndoRedoShortcuts } from '@/hooks/useUndoRedo'
import { VersionManager } from '@/utils/versionManager'
import ResumeScoreCard from '@/components/ResumeScoreCard'
import { Undo, Redo, History } from 'lucide-react'
// ... 其他导入

export default function EditorPage() {
  // 使用撤销/重做管理简历数据
  const {
    state: resumeData,
    setState: setResumeData,
    undo,
    redo,
    canUndo,
    canRedo,
    historySize,
    futureSize
  } = useUndoRedo<ResumeData>({
    personalInfo: {
      name: '',
      title: '',
      email: '',
      phone: '',
      location: '',
      website: '',
      summary: ''
    },
    experience: [],
    education: [],
    skills: [],
    projects: []
  }, {
    maxHistory: 50,
    onChange: (newData) => {
      // 自动保存版本
      if (VersionManager.shouldAutoSave()) {
        VersionManager.saveVersion(newData, '自动保存', true)
      }
    }
  })

  // 启用键盘快捷键
  useUndoRedoShortcuts(undo, redo, true)

  // 版本管理函数
  const handleSaveVersion = useCallback(() => {
    const note = prompt('请输入版本备注（可选）:')
    const version = VersionManager.saveVersion(
      resumeData,
      note || `版本 ${new Date().toLocaleString()}`,
      false,
      ['manual']
    )
    alert(`版本保存成功！`)
  }, [resumeData])

  const handleViewVersions = useCallback(() => {
    const versions = VersionManager.getVersions()
    const stats = VersionManager.getStats()
    alert(`共有 ${stats.total} 个版本\n最新: ${stats.latest?.toLocaleString()}`)
  }, [])

  // 加载数据
  useEffect(() => {
    try {
      const savedData = localStorage.getItem('resumeData')
      if (savedData) {
        setResumeData(JSON.parse(savedData))
      }
    } catch (error) {
      console.error('加载失败:', error)
    }
  }, [])

  // 更新函数
  const handleResumeUpdate = useCallback((newData: ResumeData) => {
    setResumeData(newData)
  }, [setResumeData])

  return (
    <div className="min-h-screen bg-gray-50">
      {/* 工具栏 */}
      <EditorToolbar {...toolbarProps} />

      {/* 撤销/重做和版本管理 */}
      <div className="flex items-center space-x-2 mb-4 mx-4">
        <div className="flex items-center space-x-1 bg-white rounded-lg border p-1">
          <button onClick={undo} disabled={!canUndo} className="btn btn-sm">
            <Undo className="w-4 h-4" />
            撤销 {historySize > 0 && `(${historySize})`}
          </button>
          <button onClick={redo} disabled={!canRedo} className="btn btn-sm">
            <Redo className="w-4 h-4" />
            重做 {futureSize > 0 && `(${futureSize})`}
          </button>
        </div>
        
        <div className="flex items-center space-x-1 bg-white rounded-lg border p-1">
          <button onClick={handleSaveVersion} className="btn btn-sm">
            <History className="w-4 h-4" />
            保存版本
          </button>
          <button onClick={handleViewVersions} className="btn btn-sm">
            历史
          </button>
        </div>
      </div>

      {/* 评分卡片 */}
      <div className="mx-4 mb-4">
        <ResumeScoreCard resumeData={resumeData} showDetails={true} />
      </div>

        />
```

## 🐛 常见问题排查

### 问题1: 撤销/重做不工作
**检查项：**
- 是否正确导入了 `useUndoRedo`
- 是否使用 `setResumeData` 而不是 `setResumeData` 的旧版本
- 是否调用了 `useUndoRedoShortcuts`

**解决方案：**
```typescript
// 确保使用解构出来的 setState
const { state, setState } = useUndoRedo(...)
setState(newData)  // ✅ 正确
setResumeData(newData)  // ❌ 错误（如果是旧的 useState）
```

### 问题2: 版本保存失败
**检查项：**
- LocalStorage 是否已满
- 数据格式是否正确

**解决方案：**
```typescript
// 清理旧版本
VersionManager.deleteAutoVersions()
```

### 问题3: 评分不显示
**检查项：**
- 组件是否正确导入
- resumeData 是否传递正确

**解决方案：**
```typescript
// 检查 console 是否有错误
console.log('Resume Data:', resumeData)
```
