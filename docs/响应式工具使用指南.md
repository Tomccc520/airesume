## 🎯 useResponsive - 响应式断点检测

### 基础使用

```typescript
import { useResponsive } from '@/hooks/useResponsive';

function MyComponent() {
  const { isMobile, isTablet, isDesktop, breakpoint } = useResponsive();
  
  return (
    <div>
      {isMobile && <MobileView />}
      {isTablet && <TabletView />}
      {isDesktop && <DesktopView />}
      <p>当前断点: {breakpoint}</p>
    </div>
  );
}
```

### 断点定义

| 断点 | 最小宽度 | 设备类型 | 说明 |
|------|----------|----------|------|
| xs | 375px | mobile | 小型手机 |
| sm | 640px | mobile | 大型手机 |
| md | 768px | tablet | 平板竖屏 |
| lg | 1024px | desktop | 平板横屏/小型笔记本 |
| xl | 1280px | desktop | 桌面显示器 |
| 2xl | 1536px | desktop | 大型显示器 |

### 高级功能

```typescript
function ResponsiveComponent() {
  const { 
    width,           // 当前屏幕宽度
    height,          // 当前屏幕高度
    isBelow,         // 检查是否小于指定断点
    isAbove,         // 检查是否大于指定断点
    isBetween        // 检查是否在指定范围内
  } = useResponsive();
  
  return (
    <div>
      {/* 小于 lg 断点时显示 */}
      {isBelow('lg') && <CompactLayout />}
      
      {/* 大于等于 md 断点时显示 */}
      {isAbove('md') && <ExpandedLayout />}
      
      {/* 在 md 到 lg 之间时显示 */}
      {isBetween('md', 'lg') && <TabletLayout />}
      
      <p>屏幕尺寸: {width} x {height}</p>
    </div>
  );
}
```

## 🔄 useOrientation - 屏幕方向检测

### 基础使用

```typescript
import { useOrientation } from '@/hooks/useOrientation';

function OrientationAware() {
  const { orientation, isLandscape, isPortrait, angle } = useOrientation();
  
  return (
    <div>
      <p>当前方向: {orientation}</p>
      <p>屏幕角度: {angle}°</p>
      {isLandscape && <p>横屏模式 - 显示更多内容</p>}
      {isPortrait && <p>竖屏模式 - 紧凑布局</p>}
    </div>
  );
}
```

### 方向类型

- **portrait** - 竖屏
- **landscape** - 横屏

### 详细方向信息

```typescript
function DetailedOrientation() {
  const { type, angle } = useOrientation();
  
  // type 可能的值:
  // - 'portrait-primary'    竖屏主方向
  // - 'portrait-secondary'  竖屏次方向（倒置）
  // - 'landscape-primary'   横屏主方向
  // - 'landscape-secondary' 横屏次方向（倒置）
  
  return <p>详细方向: {type} ({angle}°)</p>;
}
```

### 简化版 Hooks

```typescript
import { useIsLandscape, useIsPortrait } from '@/hooks/useOrientation';

// 仅判断是否为横屏
function LandscapeOnly() {
  const isLandscape = useIsLandscape();
  return isLandscape ? <WideLayout /> : <NarrowLayout />;
}

// 仅判断是否为竖屏
function PortraitOnly() {
  const isPortrait = useIsPortrait();
  return isPortrait ? <VerticalLayout /> : <HorizontalLayout />;
}
```

```typescript
    console.log('方向已变化:', orientation, type);

  return <Editor />;
}
```

### 方向锁定（实验性功能）

```typescript
import { useOrientationLock } from '@/hooks/useOrientation';

function VideoPlayer() {
  const { lock, unlock, isLocked } = useOrientationLock();
  
  const handleFullscreen = async () => {
    // 进入全屏
    await document.documentElement.requestFullscreen();
    // 锁定为横屏
    await lock('landscape');
  };
  
  const handleExitFullscreen = () => {
    unlock();
    document.exitFullscreen();
  };
  
  return (
    <div>
      <video />
      <button onClick={handleFullscreen}>全屏播放（横屏）</button>
      {isLocked && <button onClick={handleExitFullscreen}>退出全屏</button>}
    </div>
  );
}
```

## 👆 useSwipeGesture - 滑动手势检测

### 基础使用

```typescript
import { useSwipeGesture } from '@/hooks/useSwipeGesture';

function ImageGallery() {
  const [currentIndex, setCurrentIndex] = useState(0);
  
  const swipeHandlers = useSwipeGesture({
    onSwipeLeft: () => setCurrentIndex(i => i + 1),
    onSwipeRight: () => setCurrentIndex(i => Math.max(0, i - 1)),
  });
  
  return (
    <div {...swipeHandlers}>
      <img src={images[currentIndex]} alt="Gallery" />
    </div>
  );
}
```

### 滑动回调

```typescript
function SwipeDemo() {
  const handlers = useSwipeGesture({
    // 向左滑动
    onSwipeLeft: (event) => {
      console.log('向左滑动', event.distance, 'px');
    },
    
    // 向右滑动
    onSwipeRight: (event) => {
      console.log('向右滑动', event.distance, 'px');
    },
    
    // 向上滑动
    onSwipeUp: (event) => {
      console.log('向上滑动', event.distance, 'px');
    },
    
    // 向下滑动
    onSwipeDown: (event) => {
      console.log('向下滑动', event.distance, 'px');
    },
    
    // 任意方向滑动
    onSwipe: (event) => {
      console.log('滑动方向:', event.direction);
      console.log('滑动距离:', event.distance);
      console.log('滑动速度:', event.velocity);
    },
    
    // 滑动开始
    onSwipeStart: (x, y) => {
      console.log('开始滑动:', x, y);
    },
    
    // 滑动进行中
    onSwipeMove: (x, y, deltaX, deltaY) => {
      console.log('滑动中:', deltaX, deltaY);
    },
    
    // 滑动结束
    onSwipeEnd: () => {
      console.log('滑动结束');
    },
  });
  
  return <div {...handlers}>可滑动区域</div>;
}
```

### 配置选项

```typescript
function ConfiguredSwipe() {
  const handlers = useSwipeGesture(
    {
      onSwipeLeft: () => console.log('左滑'),
    },
    {
      minSwipeDistance: 100,      // 最小滑动距离（像素）
      minSwipeVelocity: 0.5,      // 最小滑动速度（像素/毫秒）
      maxSwipeTime: 500,          // 最大滑动时间（毫秒）
      preventDefault: true,       // 阻止默认行为
      stopPropagation: false,     // 阻止事件冒泡
      allowedDirections: ['left', 'right'], // 允许的滑动方向
    }
  );
  
  return <div {...handlers}>内容</div>;
}
```

## 🎨 实际应用示例

### 示例 1：响应式编辑器

```typescript
function ResponsiveEditor() {
  const { isMobile } = useResponsive();
  const { isLandscape } = useOrientation();
  
  if (isMobile && !isLandscape) {
    // 移动端竖屏：单栏布局，编辑/预览切换
    return <MobileEditor />;
  }
  
  if (isMobile && isLandscape) {
    // 移动端横屏：分屏布局
    return <SplitEditor />;
  }
  
  // 桌面端：完整编辑器
  return <DesktopEditor />;
}
```

### 示例 2：滑动切换编辑/预览

```typescript
function MobileEditor() {
  const [mode, setMode] = useState<'edit' | 'preview'>('edit');
  
  const swipeHandlers = useHorizontalSwipe(
    () => setMode('preview'),  // 左滑切换到预览
    () => setMode('edit'),     // 右滑切换到编辑
  );
  
  return (
    <div {...swipeHandlers}>
      {mode === 'edit' ? <EditView /> : <PreviewView />}
    </div>
  );
}
```

### 示例 3：自适应导航栏

```typescript
function AdaptiveNavbar() {
  const { isMobile, isTablet } = useResponsive();
  const { isLandscape } = useOrientation();
  
  // 移动端竖屏：汉堡菜单
  if (isMobile && !isLandscape) {
    return <HamburgerMenu />;
  }
  
  // 移动端横屏或平板：简化导航
  if ((isMobile && isLandscape) || isTablet) {
    return <CompactNavbar />;
  }
  
  // 桌面端：完整导航
  return <FullNavbar />;
}
```

## ⚠️ 注意事项

### 性能优化

1. **防抖处理**：`useResponsive` 内置了 150ms 的防抖，避免频繁触发
2. **避免过度使用**：在同一组件中多次调用会创建多个监听器
3. **使用简化版 Hook**：如果只需要判断设备类型，使用 `useIsMobile` 等简化版

### 服务端渲染（SSR）

所有 Hook 都支持 SSR，会在服务端返回默认值：
- `useResponsive`: 默认为桌面设备 (1024x768)
- `useOrientation`: 默认为竖屏
- `useSwipeGesture`: 在服务端不执行任何操作

### 浏览器兼容性

- **useResponsive**: 所有现代浏览器
- **useOrientation**: 现代浏览器，旧版浏览器使用降级方案
- **useSwipeGesture**: 支持触摸事件的设备
