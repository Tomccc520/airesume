## 🚀 最小化集成（3步）

### 步骤 1: 添加导入

在 `src/app/editor/page.tsx` 顶部添加：

```tsx
import EmptyState from '@/components/EmptyState'
import ExportPreviewDialog from '@/components/ExportPreviewDialog'
import { useVersionHistory } from '@/hooks/useVersionHistory'
```

### 步骤 2: 添加状态

在 `EditorPage` 组件中添加：

```tsx
// 空状态
const [showEmptyState, setShowEmptyState] = useState(true)

// 导出对话框
const [showExportDialog, setShowExportDialog] = useState(false)

// 版本历史
const {
  versions,
  currentVersionIndex,
  createVersion,
  restoreVersion,
  undo,
  redo,
  canUndo,
  canRedo
} = useVersionHistory(resumeData, {
  maxVersions: 10,
  autoSaveInterval: 60000
})

// 检查简历是否为空
const isResumeEmpty = () => {
  return !resumeData.personalInfo.name && 
         resumeData.experience.length === 0 &&
         resumeData.education.length === 0 &&
         resumeData.skills.length === 0
}
```

### 步骤 3: 更新 UI

#### 3.1 添加空状态（在编辑器区域）

找到这段代码：
```tsx
<div className="flex-1 lg:flex-1 bg-white rounded-lg border border-gray-200 shadow-sm overflow-hidden min-h-0">
  <ResumeEditor
    resumeData={resumeData}
    onUpdate={handleResumeUpdate}
    // ...
  />
</div>
```

替换为：
```tsx
<div className="flex-1 lg:flex-1 bg-white rounded-lg border border-gray-200 shadow-sm overflow-hidden min-h-0">
  {isResumeEmpty() && showEmptyState ? (
    <EmptyState
      onStartFromScratch={() => setShowEmptyState(false)}
      onUseTemplate={() => {
        // TODO: 打开模板选择器
        setShowEmptyState(false)
      }}
      onAIGenerate={() => {
        handleAIGenerateResume()
        setShowEmptyState(false)
      }}
      onImportData={() => {
        // TODO: 实现导入功能
        setShowEmptyState(false)
      }}
    />
  ) : (
    <ResumeEditor
      resumeData={resumeData}
      onUpdate={handleResumeUpdate}
      onOpenAIAssistant={handleOpenAIAssistant}
      currentTemplate={currentTemplate}
      onTemplateSelect={setCurrentTemplate}
    />
  )}
</div>
```

#### 3.2 添加导出对话框（在页面底部）

在 `</StyleProvider>` 之前添加：
```tsx
<ExportPreviewDialog
  isOpen={showExportDialog}
  onClose={() => setShowExportDialog(false)}
  onExport={(format) => {
    if (format === 'json') {
      // 导出 JSON
      const dataStr = JSON.stringify(resumeData, null, 2)
      const blob = new Blob([dataStr], { type: 'application/json' })
      const url = URL.createObjectURL(blob)
      const link = document.createElement('a')
      link.href = url
      link.download = `${resumeData.personalInfo.name || '简历'}.json`
      link.click()
      URL.revokeObjectURL(url)
    } else {
      handleExport(format)
    }
  }}
  resumeName={resumeData.personalInfo.name}
/>
```

#### 3.3 更新导出按钮

找到导出按钮的 `onClick`，改为：
```tsx
onClick={() => setShowExportDialog(true)}
```

### 添加版本历史面板

1. 导入组件：
```tsx
import VersionHistoryPanel from '@/components/VersionHistoryPanel'
```

2. 添加状态：
```tsx
const [showVersionHistory, setShowVersionHistory] = useState(false)
```

3. 在页面底部添加：
```tsx
<VersionHistoryPanel
  isOpen={showVersionHistory}
  onClose={() => setShowVersionHistory(false)}
  versions={versions}
  currentVersionIndex={currentVersionIndex}
  onRestore={(versionId) => {
    const data = restoreVersion(versionId)
    if (data) setResumeData(data)
  }}
  onDelete={(versionId) => {
    deleteVersion(versionId)
  }}
/>
```

4. 在工具栏添加按钮：
```tsx
<button
  onClick={() => setShowVersionHistory(true)}
  className="..."
>
  <Clock className="w-4 h-4" />
  版本历史
</button>
```
